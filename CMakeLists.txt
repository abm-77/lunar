cmake_minimum_required(VERSION 3.22)
project(lunar LANGUAGES C CXX)

include(ExternalProject)

set(THIRD_PARTY_DIR "${CMAKE_SOURCE_DIR}/third_party")
set(LLVM_SRC_DIR "${THIRD_PARTY_DIR}/llvm-project/llvm")
set(GLFW_SRC_DIR "${THIRD_PARTY_DIR}/glfw")

set(DEPS_PREFIX "${CMAKE_BINARY_DIR}/deps")
set(LLVM_INSTALL "${DEPS_PREFIX}/llvm")
set(GLFW_INSTALL "${DEPS_PREFIX}/glfw")

# ---- Build/install LLVM (X86 only) ----
ExternalProject_Add(llvm_ep
SOURCE_DIR "${LLVM_SRC_DIR}"
BINARY_DIR "${CMAKE_BINARY_DIR}/_b_llvm"
INSTALL_DIR "${LLVM_INSTALL}"
CMAKE_ARGS
-G Ninja
-DCMAKE_BUILD_TYPE=Release
-DCMAKE_INSTALL_PREFIX=${LLVM_INSTALL}
-DLLVM_TARGETS_TO_BUILD=X86;SPIRV
-DLLVM_ENABLE_PROJECTS=
-DLLVM_INCLUDE_TESTS=OFF
-DLLVM_INCLUDE_EXAMPLES=OFF
-DLLVM_INCLUDE_BENCHMARKS=OFF
-DLLVM_BUILD_LLVM_DYLIB=ON
-DLLVM_LINK_LLVM_DYLIB=ON
-DLLVM_INSTALL_UTILS=ON
-DCMAKE_C_COMPILER=${CMAKE_C_COMPILER}
-DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER}
)

# ---- Build/install GLFW (static, no extras) ----
ExternalProject_Add(glfw_ep
SOURCE_DIR "${GLFW_SRC_DIR}"
BINARY_DIR "${CMAKE_BINARY_DIR}/_b_glfw"
INSTALL_DIR "${GLFW_INSTALL}"
CMAKE_ARGS
-G Ninja
-DCMAKE_BUILD_TYPE=Release
-DCMAKE_INSTALL_PREFIX=${GLFW_INSTALL}
-DBUILD_SHARED_LIBS=OFF
-DGLFW_BUILD_EXAMPLES=OFF
-DGLFW_BUILD_TESTS=OFF
-DGLFW_BUILD_DOCS=OFF
-DGLFW_INSTALL=ON
-DCMAKE_C_COMPILER=${CMAKE_C_COMPILER}
-DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER}
)

# ---- Build umbral AFTER deps are installed ----
set(UMBRAL_BIN "${CMAKE_BINARY_DIR}/_b_umbral")

ExternalProject_Add(umbral_ep
SOURCE_DIR "${CMAKE_SOURCE_DIR}/umbral"
BINARY_DIR "${UMBRAL_BIN}"
DEPENDS llvm_ep glfw_ep
CMAKE_ARGS
-G Ninja
-DCMAKE_BUILD_TYPE=Debug
-DCMAKE_C_COMPILER=${CMAKE_C_COMPILER}
-DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER}
-DLLVM_DIR=${LLVM_INSTALL}/lib/cmake/llvm
-DGLFW_ROOT=${GLFW_INSTALL}
-DDEPS_VK_HEADERS=${THIRD_PARTY_DIR}/Vulkan-Headers/include
-DGOOGLETEST_DIR=${THIRD_PARTY_DIR}/googletest
-DFILECHECK=${LLVM_INSTALL}/bin/FileCheck
-DLZ4_DIR=${THIRD_PARTY_DIR}/lz4

INSTALL_COMMAND ""
)

# Convenience target so you can just build "engine"
add_custom_target(umbral DEPENDS umbral_ep)
